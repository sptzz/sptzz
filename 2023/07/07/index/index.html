JavaScript_day01

1.基本语法：

1.1 页面弹出警示框
alert('你好，世界')
1.2 使用JavaScript向输入内容
document.write('我是一段文字')
document.write('<h1>我是一级标题</h1>')
1.3 控制台调试，程序员自用
console.log('看看对不对')
1.4 显示一个对话框，对话框中显示一段文字，提示用户输入文字
prompt('请输入你的姓名')
1.5 变量初始化
// 声明一个变量并赋值
let age = 18
console.log(age)

// 声明多个变量
let age = 18, uname = 'sptz'
//并不推荐如此连写
1.6 变量命名规则与规范

规则：

  1. 不能用JS内置的一些英语词汇，比如：for、if、let、var等等
  2. 只能用下划线、字母、数字、＄组成且不能以数字开头
  3. 字母严格区分大小写，如Age和age是不同的变量

规范：

  1. 起名要有意义
  2. 遵守小驼峰命名法：第一个单词首字母小写，后面每个单词首字母大写，如：userName
1.7 数组
// 声明数组
let arr = ['我可以','斑马斑马','青梅','蝴蝶之夏','云烟成雨']
// 打印数组的2号位(实际是数组的第三个元素)
document.write(arr[2])//青梅
1.8 常量
// 常量，声明时必须赋值且不能更改
const G = 9.8
document.write(G)
1.9 字符串
let age = 18
// 1.使用'+'号将字符串相连
document.write('我今年' + age + '岁了')
// 2.外面使用反引号包裹，里面使用${}来包裹变量
document.write(`我今年${age}岁了`)
// 两种方法均能实现
1.10 布尔类型
// 字面量只能是true或者false
let isCool = true
console.log(isCool)
1.11 未定义类型
// 凡是声明未赋值的变量默认值是undefined
let num
console.log(num)
1.12 空类型
//表示给num赋值，但赋值为空
let num = null
console.log(num)
1.13 数据类型的检测
// typeof可以检测数据的类型
let num = 8
console.log(typeof num)//number
let str = 'sptz'
console.log(typeof str)//string
1.14 数据类型隐式转换
// +号只要是左右两边任意有一个字符串，就会默认将相加操作视为字符串相加
console.log(1 + 1)//2
console.log('sptz' + 1)//sptz1
// -号、*号、/号、%号只要是左右两边任意有一个数字，就会默认将视为数字操作
console.log(2 - '2')//0
console.log(4 / '2')//2
console.log('3' * 3)//9
console.log(9 % 2)//1
// 当+号作为正号存在时，可以将字符型转换为数字型
console.log(+ '123')//123
console.log(typeof + '123')//number
1.15 数据类型显式转换
// Number(数据)可以将数据转换成数字型
let str = '345'
console.log(str)//'345'
console.log(Number(str))//345
//当转换的字符串里面有非数字，则输出NaN
//NaN也是数字类型
let num = 123.45
// parseInt()只保留整数
console.log(parseInt(num))//123
// parseFloat()可以保留小数
console.log(parseFloat(num))//123.45
// parseInt()和parseInt()也能实现字符型到数据型的转换并作出取舍
console.log(parseInt('12px'))//12
console.log(parseInt('12.34px'))//12
console.log(parseFloat('12px'))//12
console.log(parseFloat('12.34px'))//12.34
// // parseInt()和parseInt()无法对复杂字符串进行取舍转换
console.log(parseInt('ab12px'))//NaN
console.log(parseFloat('ab12px'))//NaN

2.案例分析：

2.1 输入用户名，再将其打印输出
let uname = prompt('请输入你的姓名')
document.write(uname)
2.2 将num1的数值和num2的数值进行交换
let num1 = 10
let num2 = 20
let temp = num1
num1 = num2
num2 = temp
document.write(num1, num2)
2.3 弹出对话框输入圆的半径计算圆的面积
const PI = 3.14
let r = prompt('请输入圆的半径：')
let S = document.write(PI * r * r)
2.4 输入两个数，计算它们的和，并打印出来
let num1 = +prompt('请输入第一个数')
let num2 = +prompt('请输入第二个数')
alert(`两个数相加的和为：${num1 + num2}`)

3.1 利用现学知识，实现表格订单确认
<!DOCTYPE html>
<html lang="zh-CN">

        }

            margin: 0 auto;
            /* 合并相邻边框 */
            border-collapse: collapse;

            text-align: center;
        }
        
        table,th,td {
            border: 1px solid #000;
        }

        th {
            padding: 5px 30px;
        }
    </style>
</head>
<body>
    <script>
        // 用户输入
        let price = +prompt('请输入商品价格：')
        let num = +prompt('请输入商品数量：')
        let address = prompt('请输入收货地址：')
        // 计算总额
        let total = price * num 
        // 打印渲染
        document.write(`
            <table>
                <caption>
                    <h2>订单确认</h2>
                </caption>
                <thead>
                    <tr>
                        <th>商品名称</th>
                        <th>商品价格</th>
                        <th>商品数量</th>
                        <th>总价</th>
                        <th>收货地址</th>
                    </tr>
                </thead>
                <tbody>
                    <tr>
                        <td>华为14promax 1TB 暗夜紫</td>
                        <td>${price}元</td>
                        <td>${num}</td>
                        <td>${total}元</td>
                        <td>${address}</td>
                    </tr>
                </tbody>
            </table>
        `)
    </script>
</body>
</html>
