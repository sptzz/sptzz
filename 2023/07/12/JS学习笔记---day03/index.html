1.基本语法：

1.1 for循环语句
for (变量起始值; 终止条件; 变量变化量) {
    需要循环的语句
}

for (;;) {
    
}//等同于
while (true) {
    
}//都是无限循环
1.2 for循环和while循环的使用推荐

如果明确了循环次数的时候推荐使用for循环

如果不明确循环次数的时候推荐使用while循环

1.3 for循环嵌套
for (外部声明记录循环次数的变量; 循环条件; 变化值) {
	for (内部声明记录循环次数的变量; 循环条件; 变化值) {
        需要循环的语句
    }
}
1.4 数组的更改
let arr = []
arr[0] = 1
arr[1] = 3
console.log(arr)//[1,3]
1.5 数组的新增
//数组.push() 将一个或者多个元素添加到数组末尾，并返回该数组的新长度
arr.push(元素1, 元素2, ……, 元素n)

let arr = ['红','蓝']
//console.log(arr.push('绿'))//3
arr.push('绿')
console.log(arr)//['红','蓝','绿']

//数组.unshift() 将一个或者多个元素添加到数组开头，并返回该数组的新长度
arr.unshift(元素1, 元素2, ……, 元素n)

let arr = ['红','蓝']
//console.log(arr.unshift('绿'))//3
arr.unshift('绿')
console.log(arr)//['绿','红','蓝']
1.6 数组的删除
//数组.pop() 从数组中删除最后一个元素，并返回该元素的值

let arr = [3,2,5]
//console.log(arr.pop())//5
arr.pop()
console.log(arr)//[3,2] 

//数组.shift() 从数组中删除第一个元素，并返回该元素的值

let arr = [3,2,5]
//console.log(arr.pop())//3
arr.shift()
console.log(arr)//[2,5]

//数组.splice 删除指定元素
arr.splice(起始位置,删除几个元素)
//如果只写起始位置，则会从起始位置开始删除该位置以及后面的所以元素

let arr = [3,2,5]
arr.splice(1,1)
console.log(arr)//[3,5]
1.7 数组排序
let arr = [3,4,1,2,5]
arr.sort()
console.log(arr)//[1,2,3,4,5]
//默认为升序排序

//升序排序
arr.sort(function (a, b){
    return a - b
})
//降序排序
arr.sort(function (a, b){
    return b - a
})

2.案例分析：

2.1 利用for循环输出1~100
for (let num = 1; num <= 100; num++) {
    document.write(`${num}<br>`)
}
2.2 利用for循环输出1~100偶数和
let sum = 0
for (let i = 1; i<= 100 ;i++) {
    if (i % 2 ===0) {
        sum = sum + i
    }
}
document.write(sum)//2550
2.3 打印数组
let arr = ['王心怡','吴佩玲','宋冬野','赵雷','陈鸿宇','谢春花']
for (let i = 0; i < arr.length; i++) {
    console.log(arr[i]);
}
2.4 利用for循环嵌套计算：假如每天记住5个单词，3天记住多少个单词
for (let i = 1; i <= 3; i++) {
    document.write(`今天是第${i}天<br>`)
    for (let j = 1; j <= 5; j++) {
        document.write(`今天记住了第${j}个单词<br>`)
    }
}
2.5 输出行列数自定义的小星星
let row = +prompt('请输入需要打印的行数')
let col = +prompt('请输入需要打印的列数')
for (let i = 1;i <= row; i++) {
    for (let j = 1; j <= col; j++) {
        document.write('★')
    }
    document.write('<br>')
}
2.6 输出行列数自定义的倒三角小星星
let rl = +prompt('请输入需要打印的行列数')
for (let i = 1;i <= rl; i++) {
    for (let j = 1; j <= i; j++) {//列数跟随行数变化
        document.write('★')
    }
    document.write('<br>')
}
2.7 数组求和
let arr = [2,6,1,7,4]
let sum = 0
for (let i = 0; i < arr.length; i++) {
    sum = sum + arr[i]
}
console.log(sum)
let ag = sum / arr.length
console.log(ag)
2.8 数组求最大值和最小值
//求最大值
let arr = [2,6,1,7,4]
let max = arr[0]
for (let i = 1; i < arr.length; i++) {
    if (max < arr[i]) {
        max = arr[i]
    }
}
console.log(max)//7

//求最小值
let arr = [2,6,1,7,4]
let min = arr[0]
for (let i = 1; i < arr.length; i++) {
    if (min > arr[i]) {
        min = arr[i]
    }
}
console.log(min)//1
2.9 给数组的每一项后面增加’色’
let arr = ['红','绿','蓝','黄']
for (let i = 0; i < arr.length; i++) {
    arr[i] = arr[i] + '色'
}
console.log(arr)//['红色','绿色','蓝色','黄色']
2.10 把数组 [2,0,6,1,77,0,52,0,25,7]中大于10的元素筛选出来放入一个新数组
let arr = [2,0,6,1,77,0,52,0,25,7]
let newarr = []
for (let i = 0; i < arr.length; i++) {
    if (arr[i] > 10) {
        newarr.push(arr[i])
    }
}
console.log(newarr)//[77,52,25]
2.11 冒泡排序
let arr = [5,4,3,2,1]
for (let i = 0; i < arr.length - 1; i++) {
    for (let j = 0; j < arr.length - i - 1; j++) {
        if (arr[j] > arr[j + 1]) {
            let temp = arr[j]
            arr[j] = arr[j + 1]
            arr[j + 1] = temp
        }
    }
}
console.log(arr)//[1,2,3,4,5]

JavaScript

let arr = []
for (let i = 1; i <= 4; i++) {
    let num = prompt(`请输入第${i}季度的数据`)
    arr.push(num)
}
document.write(`<div class="box">`)
for (let i = 0; i < arr.length; i++) {
    document.write(`
    <div style="height: ${arr[i]}px;">
            <span>${arr[i]}</span>
            <h4>第${i + 1}季度</h4>
        </div>
    `)
}
document.write(`</div>`)
