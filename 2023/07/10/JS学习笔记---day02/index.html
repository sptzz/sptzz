JavaScript学习

JavaScript_day02

1.基本语法：

1.1 赋值运算符
// +=
let num = 1
//num += 1 相当于 num = num + 1
num += 1
console.log(num)//2
//此外还有 -= 、*= 、/= 、%= 原理与 +=相同
1.2 一元运算符
let num = 1
// 让num的值加1变2
num++
console.log(num)//2
// ++num也能实现num++的功能,两者独立使用时没用区别

// 区别

// ++num
let num = 1
// num先自己加1再和2相加
let i = ++num + 2
console.log(i)//4
console.log(num)//2

// num++
let num = 1
// num先和2相加得出结果，num自己再加1，但是此时num的值不会改变原本的运算结果
let i = num++ + 2
console.log(i)//3
console.log(num)//2
1.3 比较运算符
console.log(3 > 5)//false
console.log(3 >= 3)//true
console.log(2 == 2)//true
// 双等只判断值，并具有隐式转换的功能
console.log(2 == '2')//true
//全等判断值和数据类型
console.log(2 === '2')//false
// !=只判断值，并具有隐式转换的功能
console.log(2 != '2')//false
//!==判断值和数据类型
console.log(2 !== '2')//true
1.4 逻辑运算符
// 逻辑与
console.log(true && true)//true
console.log(true && false)//false
console.log(false && false)//false
// 逻辑或
console.log(true || true)//true
console.log(true || false)//true
console.log(false || false)//false
// 逻辑非
console.log(!true)//false
console.log(!false)//true
1.5 if语句
// 当括号中为真时，执行if语句
// 所有的常数(除了0)都为真，所有的字符串(除了空字符串)都为真
if (2) {
    console.log('执行if语句');
}
1.6 三元运算符
//条件 ? 满足条件执行的代码 : 不满足条件执行的代码
//和双分支if语句功能相同，但更加简便 
console.log(3 > 5 ? 3 : 5)//5
1.7 switch语句
switch (数据) {
	case 值1:
        代码1
    	break
    case 值2:
        代码2
        break
    default:
        代码n
        break
}
1.8 while循环
while (循环条件) {
    要重复执行的代码
}
1.9 循环退出

break：退出循环

continue：结束本次循环，继续下次循环

//break
let i = 1
while (i <= 5) {
    if (i === 3){
        break
    }
    console.log(i)
    i++
}//1 2

//continue
let i = 1
while (i <= 5) {
    if (i === 3){
        i++
        continue
    }
    console.log(i)
    i++
}//1 2 4 5
//continue执行之后，后面的代码失效，所以需要在continue上方增加一个i++，不然会进入反复执行continue的死循环

2.案例分析：

2.1 i++和++i的综合运算
let i = 1
console.log(i++ + ++i + i)//7
//等同于1 + 3 + 3 = 7
console.log(i)//3
2.2 输入一个数，当这个数能被4整除但是不能被100整除时弹出true，否则弹出false
let num = +prompt('请输入一个数：')
alert(num % 4 === 0 && num % 100 !== 0)
2.3 账号密码登录
let uname = prompt('请输入你的用户名：')
let password = prompt('请输入你的密码：')
if (uname === 'sptz' && password === '123456') {
    alert('登录成功')
}
else {
    alert('登录失败，用户名或密码错误')
}
2.4 让用户输入年份，判断是闰年还是平年并弹框输出
let year = +prompt('请输入年份：')
if (year % 4 === 0 && year % 100 !== 0 || year % 400 === 0) {
    alert(`${year}年是闰年`)
}
else {
    alert('`${year}年是平年`')
}
2.5 判断成绩等级
let score = +prompt('请输入你的成绩：')
if (score >= 90) {
    alert('优秀')
}
else if (score < 90 && score>=70) {
    alert('良好')
}
else if (score < 70 && score >=60) {
    alert('及格')
}
else {
    alert('不及格')
}
2.6 利用三元运算符，判断输入两个数的最大值
let num1 = prompt('请输入第一个数：')
let num2 = prompt('请输入第二个数：')
alert(`更大的那个数为：${num1 > num2 ? num1 : num2}`)
2.7 利用三元运算符，给小于10 的数字前面补0
let num = prompt('请输入一个数：')
num = num < 10 ? 0 + num : num
alert(num)
2.8 简单计算器
let num1 = +prompt('请输入第一个数字：') 
let num2 = +prompt('请输入第二个数字：')
let key = prompt('请输入 + - * / 其中一个运算符：')
switch (key) {
    case '+':
        alert(num1 + num2)
        break;
    case '-':
        alert(num1 - num2)
        break
    case '*':
        alert(num1 * num2)
    case '/':
        alert(num1 / num2)
    default:
        alert('您输入的运算符不符合要求')
        break;
}
2.9 打印输出1~100
let i = 1
while (i <= 100) {
    document.write(`${i} <br>`)
    i++
}
2.10 打印输出1~100 的累加和
let i = 1
let sum = 0
while (i <= 100) {
    sum += i
    i++
}
document.write(sum)//5050
2.11 打印输出1~100的偶数和
let i = 1
let sum = 0
while (i<=100) {
   if (i % 2 === 0) {
        sum += i
   } 
   i++
}
document.write(sum)//2550

//利用三元运算符做
let i = 1
let sum = 0
while (i<=100) {
   sum = sum + (i % 2 === 0 ? i : 0)
   i++
}
document.write(sum)//2550
2.12 页面弹出对话框，’你艾希我奶妈’，如果输入’爱’，则结束，否则一直弹框
while (true) {
    let answer = prompt('你艾希我奶妈')
    if (answer === '爱') {
        break
    }
}

3. 综合案例：

3.1 简单ATM机
let money_already = 100
while (true) {
    let main = +prompt(`
    请输入对应操作的序号
    1.存钱
    2.取钱
    3.查看余额
    4.退出`)
    if(main === 4) {
        break
    }
    switch(main) {
        case 1:
        let money_increase = +prompt('请输入所存数额')
        money_already += money_increase
        break 
        case 2:
        let money_decrease = +prompt('请输入所取数额')
        money_already -= money_decrease
        break
        default:
        alert(`您的银行卡余额是：${money_already}`)
        break
    }
}
