1.基本语法：

1.1 函数
//function,是被设计为执行特定任务的代码块
function 函数名(参数列表)
    函数体
}
//函数调用,这些函数体内的代码逻辑会被执行
函数名()

function getsum(start = 0, end = 0) {//形参
    let sum = 0
    for (let i = start; i <= end; i++) {
        sum += i
    }
    document.write(sum)
}
let num1 = +prompt('请输入起始值')
let num2 = +prompt('请输入结束值')
getsum(num1, num2)
getsum(1, 100)//5050//实参
getsum(2,50)//1274//实参

//如果不输入实参，则会将形参的值视为undefined
//可以给形参赋值，使在缺少实参参数传递时，成为默认值,实参可以是变量

//当调用某个函数，这个函数会返回一个结果出来
//return后的代码将不会被执行

console.log(re)//20

1.6 逻辑中断
function fn(x,y) {
    x = x || 0
    y = y || 0//与直接给形参赋默认值为0的功能相同
    console.log(x + y)
}
fn()//0

let age = 18
console.log(false && age++)//false
//当与操作前面是false的时候，后面的代码将直接略过，不再执行
//同理，当或操作前面是true的时候，后面的代码也将直接略过，不再执行
console.log(age)//18

console.log(11 && 33)//33
//当与操作前后都为true时，会返回后一个值
console.log(11 || 33)//11
//当或操作前后都为true时，会返回第一个值
1.7 转换为布尔型
//'',0,undefined,null,false,NaN转换为布尔值都为false，其余为true

console.log(undefined && 20)//undefined
console.log(null && 20)//null 

//'-'号会是空字符串''转换为0
//null经过数字转换之后会变成0
//undefined经过数字转换之后会变为NaN

2.案例分析：

2.1 利用函数数组求和
function getsum(arr = []) {
    let sum = 0
    for (let i = 0; i <arr.length; i++) {
        sum += arr[i]
    }
    console.log(sum)
}
getsum([1,2,3,4,5,6])//21
2.2 求两个数中的最大值
function getmax(num1, num2) {
    if (num1 - num2 > 0) {
        return num1
    }
    else {
        return num2
    }
} 
let max = getmax(9, 3)
console.log(max)
2.3 求数组中的最大值和最小值
function getmax(arr = []) {
    let max = arr[0]
    let min = arr[0]
    for (let i = 1; i < arr.length; i++) {
        if (max < arr[i]) {
            max = arr[i]
        }
        if (min > arr[i]) {
            min = arr[i]
        }
    }
    return [max, min]
}
let newarr = getmax([1,3,7,45,787,23,56,166])
console.log(`数组的最大值是${newarr[0]}`)
console.log(`数组的最小值是${newarr[1]}`)

3.综合案例：

3.1 输入秒数转换成多少小时多少分钟多少秒
function time(t) {
let h = parseInt(t / 60 /60 % 24)
if (h < 10) {
    h = '0' + h
}
let m = parseInt(t / 60 % 60)
if (m < 10) {
    m = '0' + m
}
let s = parseInt(t % 60)
if (s < 10) {
    s = '0' + s
}
document.write(`${h}小时${m}分钟${s}秒`)
}
let t = +prompt('请输入秒数：')
time(t)
