1.基本语法：

1.1 let和const
//let一般用于变化的变量
//const一般用于不变化的变量
//推荐声明时使用const，如果后续发生改变再改为let
//特例
const arr = ['red', 'blue']
arr.push('green')
console.log(arr)
//虽然数组长度改变，但是存放数组的地址没有改变，所以可以使用const，同理对象更改也可以使用const

let names = []
names = [1, 2, 3]
//或者
let obj = {}
obj = {
    uname:'SPTZ'
}//此时必须使用let，因为此时视为重新创建了一个数组或一个对象，地址已经改变
1.2 DOM
//DOM是将标签获取到JS里面，以对象的格式储存 
//获取DOM元素
document.querySelector('CSS选择器(和CSS选择标签格式相同)')
//如果没有取到，则返回null 
//选择匹配的第一个元素，只能选择一个！，可以对获取过来的标签直接进行修改
//比如
<div>雪之下雪乃</div>
const div = document.querySelector('div')
div.style.color = 'red'
//div标签中的文本由默认颜色变为红色

document.querySelectorAll('CSS选择器(和CSS选择标签格式相同)')
//加上了All,可以选择所以符合条件的标签，并把标签放在一个伪数组里面，就算只获取过来一个元素，也是伪数组
//伪数组有长度和索引号，但是不能进行pop(),push()等数组操作
//因为是数组，所以不能直接对获取过来的标签进行修改，需要使用循环遍历的方法进行修改
//或者，使用数组下标的方式修改

//了解即可
//根据id获取一个元素
document.getElementById('nav')
//根据标签获取一类元素 比如获取页面所以div
document.getElementsByTagName('div')
//根据类名获取元素 比如获取页面所有类名为hd的元素
document.getElementsByClassName('hd')
//打印对象专用,能够更加直观的观察
console.dir()
1.3 修改对象内容
//修改文字内容 对象.innerText 属性
<div class="box">我是文字内容</div>
<script>
	const box = document.querySelector('.box')
	box.innerText = '<strong>我是一个盒子</strong>'
	//此时页面上的内容被修改为‘<strong>我是一个盒子<strong>’
	//只能更改文本，无法解析标签
</script>

//解析标签 元素.innerHTML
//具有innerText的功能，但是能识别标签
box.innerHTML = '<strong>我是一个盒子</strong>'
//此时页面上的文本被更改为加粗的‘我是一个盒子’
//更加推荐使用innerHTML

//更改标签属性
<img src="../img/岩永琴子.jpg" alt="">
<script>
    const img = document.querySelector('img')
    img.src = '../img/秋天.jpg'
</script>
//此时，页面上的图片会被更改为秋天.jpg

//更改style属性
<style>
    .box {
        width: 200px;
        height: 200px;
        background-color: aqua;
    }
</style>
    <div class="box"></div>
<script>
    const box = document.querySelector('.box')
    box.style.with = '300px'
	box.style.backgroundColor = 'red'
</script>

//通过类名修改
//元素.className = '类名'
<style>
    div {
        width: 200px;
        height: 200px;
        background-color: red;
    }

    .box {
        width: 300px;
        height: 300px;
        background-color: skyblue;
        margin: 100px auto;
        padding: 10px;
        border: 1px solid black;
    }
</style>
<div></div>
<script>
    const div = document.querySelector('div')
    div.className = 'box'
</script>
//会覆盖掉原有属性，如果需要保留，应该在className将原有类名加上
//如果需要增加多个类，元素.className = '类名1 类名2'

//最好用的方法
//通过classList操作类控制CSS
//追加一个类
元素.classList.add('类名')
//删除一个类
元素.classList.remove('类名')
//切换一个类 有就删掉，没有就加上
元素.classList.toggle('类名')
<style>

        color: black;
    }    

    .active {
        color: red;

    }
</style>
<div class="box">逢坂大河</div>

    const div = document.querySelector('div')
    div.classList.add('active')

1.4 获取设置表单的值
<input type = "text" value = "麻辣烫">
<script>
    const uname = document.querySelector('input')
    //表单元素只能通过value值去获取
	console.log(uname.value)//麻辣烫
	console.log(uname.innerHTML)//空白，无法得到表单内容
	//表单元素只能通过value值去更改
	uname.value = '我要吃麻辣烫'
	console.log(uname.value)//我要吃麻辣烫
	//表单可以更改type值
	//通过更改type值在text和password之间切换可以实现密码的显示和隐藏
	uname.type = 'password'
</script>

//复选框的勾选与取消
<input type="checkbox">
<script>
    const input = document.querySelector('input')
    console.log(input.checked)//false
    input.checked = true
    console.log(input.checked)//true
</script>
1.5 自定义属性
//data-自定义
<div data-id="1" data-num="one" data-spm="哈哈哈">1</div>
<div>2</div>
<div>3</div>
<div>4</div>
<div>5</div>

    const one  = document.querySelector('div')
    console.log(one.dataset)//将div的全部属性打印

    console.log(one.dataset.spm)//哈哈哈
</script>
1.6 定时器-间歇函数
//开启定时器
setInterval(函数, 间隔时间)
//作用:每隔一段时间调用这个函数
//间隔时间单位是毫秒

<script>
    setInterval(function () {
        document.write('中野一花<br>')
    }, 1000)
</script>
//每隔一秒钟打印一次“中野一花”
let n = setInterval(function () {
    document.write('中野二乃<br>')
}, 1000)
console.log(n)//2

let m = setInterval(function () {
    document.write('中野三玖<br>')
}, 1000)
console.log(m)//3
//n和m代表的2和3指这个页面中执行过的定时器的序号,可以根据序号指定开关定时器

//关闭定时器
let 变量名 = setInterval(函数, 间隔时间)
clearInterval(变量名)

3.1 随机抽取获奖人员

HTML

<div class="box">
      <h2>获奖名单</h2><br>
      <h1>一等奖：<span id="one">???</span></h1>
      <h2>二等奖：<span id="two">???</span></h2>
      <h3>三等奖：<span id="three">???</span></h3>
</div>

CSS

<style>
    .box {
        width: 500px;
        height: 330px;
        background-color: aquamarine;
        margin: 0 auto;
    }

    h1,
    h2,
    h3 {
        text-align: center;
    }
</style>

JavaScript

<script>
    // const box = document.querySelector('.box')
	// box.innerHTML = '<strong>我是一个盒子</strong>'
    const arr = ['赵雷','宋冬野','王心怡','谢春花','陈鸿宇']
    const random = Math.floor(Math.random() * arr.length)
    const one = document.querySelector('#one')
    one.innerHTML = arr[random]
    arr.splice(random, 1)

    const random2 = Math.floor(Math.random() * arr.length)
    const two = document.querySelector('#two')
    two.innerHTML = arr[random2]
    arr.splice(random2, 1)
    
    const random3 = Math.floor(Math.random() * arr.length)
    const three = document.querySelector('#three')
    three.innerHTML = arr[random3]
    arr.splice(random3, 1)
</script>

<script>
    const sliderData = [
        {url: '../LBT/1.png',title: '中野一花',color: 'rgb(237, 147, 185)'},
