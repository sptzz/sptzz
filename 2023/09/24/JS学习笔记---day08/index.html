1.基本语法：

1.1 事件流
//事件流指的是事件完整执行过程中的流动路径 
//捕获阶段 document html body div(实际比较少用)
DOM.addEventListrner(事件类型, 事件处理函数, 是否使用捕获机制true)
//当一个元素触发事件后，会依次向下调用所有子级元素的同名元素
//冒泡阶段 div html body document
DOM.addEventListrner(事件类型, 事件处理函数, 是否使用捕获机制false)
//当一个元素触发事件后，会依次向上调用所有父级元素的同名元素
//true就是捕获阶段触发，false就是冒泡阶段触发，默认是false
//事件冒泡是默认存在的

//阻止冒泡
事件对象.stopPropagation()
//阻止链接或者表单提交
e.preventDefault()

	<form action="https://sptzz.icu">
        <input type="submit" value="点击进入">
    </form>
    <script>
        const form = document.querySelector('form')
        form.addEventListener('submit', function (e) {
            e.preventDefault()
        })
    </script>
1.2 解绑事件
//绑定事件
btn.onclick = function () {
    alert('点击了')
}
//或者
btn.addEventListrner('click', function(){
    alert('点击了')
})

//解绑事件
btn.onclick = null
//或者
btn.removeEventListrner('click', fn)
//匿名函数无法解绑，解绑前需要绑定的匿名函数提出来命名

//注意
//鼠标经过事件
mouseover和mouseout会有冒泡效果
mouseenter和mouseleave没有冒泡效果(推荐)
1.3 事件委托
//利用事件冒泡的原理，给父元素注册事件，当我们触发子元素时会冒泡到父元素从而触发父元素的事件
事件对象.target.tagName可以获得真正触发事件的元素

	<ul>
        <li>第1个子元素</li>
        <li>第2个子元素</li>
        <li>第3个子元素</li>
        <li>第4个子元素</li>
        <li>第5个子元素</li>
        <p>我不需要变色</p>
    </ul>
    <script>
        const ul = document.querySelector('ul')
        ul.addEventListener('click', function(e){
            if (e.target.tagName === 'LI') {
                e.target.style.color = 'red'
                //e.target就是点击的对象
            }
        })
    </script>
1.4 页面加载事件
window.addEventListener('load', function(){
    //执行的操作 
})
//如果将JS写在head标签中，需要将JS代码写入该代码中，表示当页面资源加载完毕后回头调用函数

//如果需要加载的图片内存很大
img.addEventListener('load', function(){
    //等待图片加载完毕再执行操作 
})

//当初始的HTML文档完全加载和解析后DOMContentLoaded事件被触发，无需等待样式表和图片加载完毕
document.addEventListener('DOMContentLoaded', function(){
    //执行的操作
})
1.5 元素滚动事件
window.addEventListener('scroll', function(){
    //执行的操作        
})

scrollLeft, scrollTop
//获取元素内容往左、往上滚出去看不到的距离，可以赋值是数字型但没有单位默认为px
//document.documentElement.scrollTop = 0 可以实现回到顶部的功能

window.addEventListener('scroll', function(){
    console.log(document.documentElement.scrollTop)
	//输出向下滚动了多少像素
})

//滚动到页面指定坐标
window.scrllTo(0,0)
1.6 页面尺寸事件
//浏览器窗口大小发生变化的时候发生的事件
window.addEventListener('resize', function(){
    //执行的操作        
})

//获取元素的可见部分宽高(不包含边框，margin，滚动条)(包含padding)
clientWidth, clientHeight

//获取元素的宽高(包含边框，margin，padding)
offsetWidth, offsetHeight

//获取元素的位置(距离自己定位父级元素的左、上距离)(只读)
offsetLeft, offsetTop

//获取元素的大小及其相对于视口的位置
element.getBoundingClientRect()
